<!-- API contracts from Plan 01 -->

GET /api/portal/inbox/linkedin/conversations
Response: {
  conversations: Array<{
    id: string;               // internal cuid — URL param for conversation view
    conversationId: string;   // LinkedIn's ID — for worker calls only
    participantName: string | null;
    participantHeadline: string | null;
    participantProfilePicUrl: string | null;
    lastMessageSnippet: string | null;
    lastActivityAt: string;   // ISO timestamp
    unreadCount: number;
    jobTitle: string | null;  // from Person or participantHeadline
    company: string | null;   // from Person
    senderId: string;
  }>
}

GET /api/portal/inbox/linkedin/conversations/[conversationId]/messages
URL param conversationId = internal cuid (LinkedInConversation.id)
Query params: ?refresh=true (optional — forces worker re-fetch)
Response: {
  messages: Array<{
    id: string;
    eventUrn: string;
    senderUrn: string;
    senderName: string | null;
    body: string;
    isOutbound: boolean;
    deliveredAt: string;  // ISO timestamp
  }>;
  conversationId: string;
  participantName: string | null;
  participantUrn: string | null;
  senderId: string;
}

POST /api/portal/inbox/linkedin/reply
Body: { conversationId: string, message: string }
Response: { actionId: string }  (status 201)
Error responses: 400 (missing fields), 404 (not found), 422 (no personId)

GET /api/portal/inbox/linkedin/actions/[actionId]/status
Response: { status: string, completedAt: string | null }

POST /api/portal/inbox/linkedin/sync (EXISTING — Phase 34)
Response: { conversations: [...], lastSyncedAt: string | null, syncing: boolean }
Status: 202 if syncing, 200 if cooldown
