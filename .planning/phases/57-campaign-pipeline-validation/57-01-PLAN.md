```typescript
export interface ListValidationResult {
  valid: boolean;
  hardFailures: string[];  // blocking issues
  softWarnings: string[];  // non-blocking warnings
}

export interface DataQualityReport {
  totalPeople: number;
  withFirstName: number;
  withCompany: number;
  firstNameAndCompanyPct: number;
  channelReport: {
    channel: "email" | "linkedin";
    eligible: number;
    ineligible: number;
    ineligibleReasons: string[];
  }[];
  warnings: string[];
  pass: boolean;  // true if channel requirements met and >= 80% have name+company
}

// Validates that people in a list meet the channel requirements
export function validateListForChannel(
  channel: "email" | "linkedin",
  people: Array<{
    email?: string | null;
    linkedinUrl?: string | null;
    firstName?: string | null;
    lastName?: string | null;
    company?: string | null;
    jobTitle?: string | null;
  }>,
): ListValidationResult;

// Runs full data quality pre-check on a list for given channels
export function runDataQualityPreCheck(
  channels: ("email" | "linkedin")[],
  people: Array<{
    email?: string | null;
    linkedinUrl?: string | null;
    firstName?: string | null;
    lastName?: string | null;
    company?: string | null;
    jobTitle?: string | null;
  }>,
): DataQualityReport;
```
